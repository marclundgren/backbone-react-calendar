/**
 * @jsx React.DOM
 */

var simpleToggle  = React.createClass({
  getDefaultProps: function() {
    return {
      message: 'Tap Me'
    };
  },

  getInitialState: function() {
    return {
      active: true,
      count: 0
    };
  },

  onClick: function() {
    this.setState({
      active: !this.state.active,
      count: this.state.count + 1
    });
  },

  render: function () {
    var active = this.state.active;
    var className = active ? 'item active' : 'item';

    return (
      <div
        onClick={this.onClick}
        className={className}>
        {this.state.count || this.props.message}
      </div>
    );
  }
});

React.renderComponent(
  simpleToggle({}),
  document.getElementById('simpleToggle')
);
